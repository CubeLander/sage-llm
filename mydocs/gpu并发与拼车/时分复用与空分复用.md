## 🔀 平着吃 vs 竖着吃 - 两种策略详解

### 策略A：平着吃（时分复用）🍰➡️

```python
class HorizontalEatingStrategy:
    """
    平着吃策略 - 时分复用，一次一层
    """
    
    def __init__(self, cake_layers):
        self.layers = cake_layers
        self.gpu_sms = 108
        
    def simulate_horizontal_execution(self):
        """
        模拟平着吃的执行过程
        """
        print("🍴 平着吃策略 (时分复用)")
        print("=" * 40)
        
        total_time = 0
        cache_state = {'current_data': None, 'hit_rate': 0}
        
        execution_timeline = []
        
        for i, layer in enumerate(self.layers):
            print(f"\n⏰ 时间段 {i+1}: 执行{layer['layer']}")
            
            # 计算SM利用率
            sm_utilization = min(layer['sm_requirement'] / self.gpu_sms, 1.0)
            
            # 计算Cache效果
            cache_effect = self.calculate_cache_effect_horizontal(layer, cache_state)
            
            # 实际执行时间（考虑cache miss惩罚）
            actual_time = layer['execution_time'] * cache_effect['time_multiplier']
            
            execution_timeline.append({
                'layer': layer['layer'],
                'time_start': total_time,
                'time_end': total_time + actual_time,
                'sm_utilization': sm_utilization,
                'cache_hit_rate': cache_effect['hit_rate']
            })
            
            print(f"  🔧 SM利用率: {sm_utilization:.1%}")
            print(f"  💾 Cache命中率: {cache_effect['hit_rate']:.1%}")
            print(f"  ⏱️ 实际执行时间: {actual_time:.1f}ms")
            print(f"  🚦 GPU状态: 其他{self.gpu_sms - layer['sm_requirement']}个SM空闲")
            
            total_time += actual_time
            cache_state = cache_effect['new_state']
        
        print(f"\n📊 平着吃总结：")
        print(f"  ⏰ 总执行时间: {total_time:.1f}ms")
        print(f"  📈 平均SM利用率: {self.calculate_average_utilization(execution_timeline):.1%}")
        print(f"  💾 整体Cache效率: {self.calculate_overall_cache_efficiency(execution_timeline):.1%}")
        
        return {
            'total_time': total_time,
            'timeline': execution_timeline,
            'strategy': 'horizontal'
        }
    
    def calculate_cache_effect_horizontal(self, current_layer, cache_state):
        """
        计算平着吃时的cache效果
        """
        if cache_state['current_data'] is None:
            # 第一层，冷启动
            return {
                'hit_rate': 0.1,  # 只有10%命中率
                'time_multiplier': 1.3,  # 30%时间惩罚
                'new_state': {'current_data': current_layer['kernel_type'], 'hit_rate': 0.1}
            }
        elif cache_state['current_data'] == current_layer['kernel_type']:
            # 相同类型kernel，cache很友好
            return {
                'hit_rate': 0.9,
                'time_multiplier': 1.0,
                'new_state': {'current_data': current_layer['kernel_type'], 'hit_rate': 0.9}
            }
        else:
            # 不同类型kernel，cache基本失效
            return {
                'hit_rate': 0.15,
                'time_multiplier': 1.4,  # 40%时间惩罚
                'new_state': {'current_data': current_layer['kernel_type'], 'hit_rate': 0.15}
            }

horizontal_strategy = HorizontalEatingStrategy(cake_layers)
horizontal_result = horizontal_strategy.simulate_horizontal_execution()
```

### 策略B：竖着吃（空分复用）🍰⬆️

```python
class VerticalEatingStrategy:
    """
    竖着吃策略 - 空分复用，同时吃多层
    """
    
    def __init__(self, cake_layers):
        self.layers = cake_layers
        self.gpu_sms = 108
        
    def simulate_vertical_execution(self):
        """
        模拟竖着吃的执行过程
        """
        print("🍴 竖着吃策略 (空分复用)")
        print("=" * 40)
        
        # 检查资源可行性
        total_sm_needed = sum(layer['sm_requirement'] for layer in self.layers)
        
        if total_sm_needed > self.gpu_sms:
            print("❌ 资源不足，无法同时执行所有层")
            return self.optimize_vertical_allocation()
        
        print("✅ 资源充足，可以同时执行所有层")
        
        # 分配SM资源
        sm_allocation = self.allocate_sms_to_layers()
        
        # 计算并发执行效果
        concurrent_effects = self.calculate_concurrent_effects(sm_allocation)
        
        # 执行时间由最慢的层决定
        max_execution_time = max(effect['actual_time'] for effect in concurrent_effects)
        
        print(f"\n🎯 SM分配方案：")
        for i, (layer, allocation) in enumerate(zip(self.layers, sm_allocation)):
            effect = concurrent_effects[i]
            print(f"  {layer['layer']}")
            print(f"    分配SM: {allocation['allocated_sms']}")
            print(f"    Cache竞争惩罚: +{(effect['time_multiplier']-1)*100:.0f}%")
            print(f"    实际执行时间: {effect['actual_time']:.1f}ms")
        
        print(f"\n📊 竖着吃总结：")
        print(f"  ⏰ 总执行时间: {max_execution_time:.1f}ms")
        print(f"  📈 总体SM利用率: {total_sm_needed/self.gpu_sms:.1%}")
        print(f"  💾 Cache竞争程度: {self.assess_cache_contention(concurrent_effects)}")
        
        return {
            'total_time': max_execution_time,
            'sm_allocation': sm_allocation,
            'concurrent_effects': concurrent_effects,
            'strategy': 'vertical'
        }
    
    def allocate_sms_to_layers(self):
        """
        为各层分配SM资源
        """
        allocations = []
        current_sm = 0
        
        for layer in self.layers:
            allocation = {
                'layer_name': layer['layer'],
                'requested_sms': layer['sm_requirement'],
                'allocated_sms': layer['sm_requirement'],
                'sm_range': (current_sm, current_sm + layer['sm_requirement'] - 1)
            }
            allocations.append(allocation)
            current_sm += layer['sm_requirement']
        
        return allocations
    
    def calculate_concurrent_effects(self, allocations):
        """
        计算并发执行的效果
        """
        effects = []
        
        for i, (layer, allocation) in enumerate(zip(self.layers, allocations)):
            # Cache竞争分析
            cache_contention = self.analyze_cache_contention(layer, self.layers)
            
            # 内存带宽竞争
            bandwidth_contention = self.analyze_bandwidth_contention(layer, self.layers)
            
            # 综合时间惩罚
            time_multiplier = 1.0 + cache_contention + bandwidth_contention
            
            actual_time = layer['execution_time'] * time_multiplier
            
            effects.append({
                'layer_index': i,
                'cache_contention': cache_contention,
                'bandwidth_contention': bandwidth_contention,
                'time_multiplier': time_multiplier,
                'actual_time': actual_time
            })
        
        return effects
    
    def analyze_cache_contention(self, current_layer, all_layers):
        """
        分析Cache竞争程度
        """
        total_cache_footprint = sum(layer['cache_footprint'] for layer in all_layers)
        cache_size_mb = self.gpu_resources['l2_cache_size'] / (1024 * 1024)
        
        if total_cache_footprint <= cache_size_mb:
            return 0.0  # 无竞争
        else:
            # Cache溢出，性能下降
            overflow_ratio = total_cache_footprint / cache_size_mb
            return min(0.5, (overflow_ratio - 1) * 0.3)  # 最多50%惩罚
    
    def analyze_bandwidth_contention(self, current_layer, all_layers):
        """
        分析内存带宽竞争
        """
        # 不同内存访问模式的竞争程度不同
        memory_patterns = [layer['memory_pattern'] for layer in all_layers]
        
        if len(set(memory_patterns)) == 1:
            return 0.1  # 相同模式，轻微竞争
        else:
            return 0.2  # 不同模式，中等竞争

vertical_strategy = VerticalEatingStrategy(cake_layers)
vertical_result = vertical_strategy.simulate_vertical_execution()
```

```python
        性能对比分析

        print("-" * 30)
        
        h_time = self.horizontal['total_time']
        v_time = self.vertical['total_time']
        speedup = h_time / v_time if v_time < h_time else v_time / h_time

        print(f"🍰 竖着吃总时间: {v_time:.1f}ms")
        
        if v_time < h_time:
            print(f"🏆 竖着吃获胜！提速 {speedup:.1f}x")
            winner = "vertical"
        else:
            print(f"🏆 平着吃获胜！提速 {speedup:.1f}x")
            winner = "horizontal"
        
        return winner
```

```python
class IntelligentCakeEatingDecision:
    """
    智能蛋糕吃法决策器 - 根据实际情况选择最优策略
    """
    
    def __init__(self):
        self.decision_factors = [
            'total_workload_size',
            'cache_contention_level', 
            'latency_requirements',
            'gpu_utilization_target',
            'power_constraints'
        ]
    
    def make_decision(self, cake_layers, constraints):
        """
        智能决策：选择最优的蛋糕吃法
        """
        print("🧠 智能蛋糕吃法决策")
        print("=" * 40)
        
        # 分析输入
        workload_analysis = self.analyze_workload(cake_layers)
        
        # 评估约束
        constraint_scores = self.evaluate_constraints(constraints)
        
        # 计算决策得分
        decision_scores = self.calculate_decision_scores(
            workload_analysis, constraint_scores
        )
        
        # 做出最终决策
        final_decision = self.finalize_decision(decision_scores)
        
        return final_decision
    
    def analyze_workload(self, layers):
        """
        分析工作负载特性
        """
        total_sm_need = sum(layer['sm_requirement'] for layer in layers)
        total_cache_need = sum(layer['cache_footprint'] for layer in layers)
        execution_time_variance = np.var([layer['execution_time'] for layer in layers])
        
        return {
            'sm_requirement': total_sm_need,
            'cache_pressure': total_cache_need,
            'time_variance': execution_time_variance,
            'layer_count': len(layers)
        }
    
    def calculate_decision_scores(self, workload, constraints):
        """
        计算各种策略的得分
        """
        scores = {
            'horizontal': 0,
            'vertical': 0,
            'hybrid': 0  # 混合策略
        }
        
        # 基于工作负载的得分调整
        if workload['sm_requirement'] > 90:
            scores['vertical'] += 20  # 高SM需求适合并行
        else:
            scores['horizontal'] += 15  # 低SM需求适合顺序
        
        if workload['cache_pressure'] > 35:  # MB
            scores['horizontal'] += 25  # 高缓存压力适合顺序
        else:
            scores['vertical'] += 20
        
        # 基于约束的得分调整
        if constraints.get('latency_priority', False):
            scores['horizontal'] += 30
        
        if constraints.get('throughput_priority', False):
            scores['vertical'] += 30
        
        return scores
    
    def finalize_decision(self, scores):
        """
        最终决策
        """
        best_strategy = max(scores.keys(), key=lambda k: scores[k])
        best_score = scores[best_strategy]
        
        print(f"📊 决策得分：")
        for strategy, score in scores.items():
            marker = "🏆" if strategy == best_strategy else "  "
            print(f"  {marker} {strategy}: {score}分")
        
        decision_map = {
            'horizontal': {
                'strategy': '🍰 平着吃（时分复用）',
                'reasoning': 'Cache效率优先，延迟敏感',
                'implementation': 'Sequential kernel execution with cache optimization'
            },
            'vertical': {
                'strategy': '⚡ 竖着吃（空分复用）',
                'reasoning': '吞吐量优先，资源充分利用',  
                'implementation': 'Concurrent kernel execution with SM partitioning'
            },
            'hybrid': {
                'strategy': '🎭 混合策略（智能切换）',
                'reasoning': '根据实时负载动态调整',
                'implementation': 'Adaptive switching between strategies'
            }
        }
        
        chosen_strategy = decision_map[best_strategy]
        
        print(f"\n🎯 最终决策: {chosen_strategy['strategy']}")
        print(f"📝 决策理由: {chosen_strategy['reasoning']}")
        print(f"⚙️ 实现方案: {chosen_strategy['implementation']}")
        
        return chosen_strategy
```
