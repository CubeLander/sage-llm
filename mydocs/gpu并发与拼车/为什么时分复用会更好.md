class CacheMissExperiment:
    """
    Cache Miss实验 - 测试SM分配对缓存性能的实际影响
    """
    
    def __init__(self):
        self.l1_cache_size_per_sm = 128 * 1024  # 128KB L1 per SM
        self.l2_cache_size_total = 40 * 1024 * 1024  # 40MB L2 total
        
    def experiment_random_vs_controlled_allocation(self):
        """
        实验：随机分配 vs 受控分配的缓存性能对比
        """
        print("🔬 Cache Miss实验：随机 vs 受控分配")
        print("=" * 50)
        
        # 实验场景设定
        experiment_setup = {
            'models': ['MobileNet', 'ResNet', 'BERT'],
            'batch_sizes': [32, 64, 32],
            'cache_footprints': [8, 15, 12],  # MB per model
            'sm_requirements': [32, 48, 28]
        }
        
        print("🎭 实验场景设定：")
        for i, model in enumerate(experiment_setup['models']):
            print(f"  {model}: 批量{experiment_setup['batch_sizes'][i]}, "
                  f"Cache占用{experiment_setup['cache_footprints'][i]}MB, "
                  f"需要{experiment_setup['sm_requirements'][i]}个SM")
        
        # 场景A: 随机SM分配（当前CUDA默认行为）
        random_results = self.simulate_random_allocation(experiment_setup)
        
        # 场景B: 受控SM分配（理想情况）
        controlled_results = self.simulate_controlled_allocation(experiment_setup)
        
        # 结果对比
        self.compare_results(random_results, controlled_results)
        
    def simulate_random_allocation(self, setup):
        """
        模拟随机SM分配的缓存表现
        """
        print(f"\n🎲 场景A: 随机SM分配")
        print("-" * 25)
        
        # 模拟随机分配导致的cache冲突
        cache_conflicts = self.calculate_random_cache_conflicts(setup)
        
        results = {
            'allocation_type': 'random',
            'l1_hit_rates': [],
            'l2_hit_rates': [],
            'execution_times': [],
            'cache_conflicts': cache_conflicts
        }
        
        for i, model in enumerate(setup['models']):
            # 随机分配导致的cache命中率下降
            base_l1_hit_rate = 0.9  # 基础L1命中率90%
            base_l2_hit_rate = 0.95 # 基础L2命中率95%
            
            # 随机分配的惩罚
            l1_penalty = cache_conflicts['l1_conflicts'][i] * 0.3
            l2_penalty = cache_conflicts['l2_conflicts'][i] * 0.2
            
            actual_l1_hit = max(0.3, base_l1_hit_rate - l1_penalty)
            actual_l2_hit = max(0.6, base_l2_hit_rate - l2_penalty)
            
            # 计算执行时间（包含cache miss惩罚）
            base_time = 2.0 + i * 0.8  # 基础执行时间
            cache_miss_penalty = (1 - actual_l1_hit) * 100 + (1 - actual_l2_hit) * 300
            actual_time = base_time + cache_miss_penalty * 0.01
            
            results['l1_hit_rates'].append(actual_l1_hit)
            results['l2_hit_rates'].append(actual_l2_hit)
            results['execution_times'].append(actual_time)
            
            print(f"  {model}: L1命中{actual_l1_hit:.1%}, L2命中{actual_l2_hit:.1%}, 时间{actual_time:.1f}ms")
        
        total_time = max(results['execution_times'])
        print(f"  🕐 总执行时间: {total_time:.1f}ms (并发执行)")
        
        return results
        
    def simulate_controlled_allocation(self, setup):
        """
        模拟受控SM分配的缓存表现
        """
        print(f"\n🎯 场景B: 受控SM分配")
        print("-" * 25)
        
        results = {
            'allocation_type': 'controlled',
            'l1_hit_rates': [],
            'l2_hit_rates': [],
            'execution_times': [],
            'cache_conflicts': {'l1_conflicts': [0, 0, 0], 'l2_conflicts': [0.1, 0.1, 0.1]}
        }
        
        for i, model in enumerate(setup['models']):
            # 受控分配的优化效果
            base_l1_hit_rate = 0.95  # 更高的基础命中率
            base_l2_hit_rate = 0.98
            
            # 最小的竞争惩罚
            l1_penalty = 0.05  # 只有5%的轻微竞争
            l2_penalty = 0.02  # L2竞争很小
            
            actual_l1_hit = base_l1_hit_rate - l1_penalty
            actual_l2_hit = base_l2_hit_rate - l2_penalty
            
            # 执行时间优化
            base_time = 2.0 + i * 0.8
            cache_miss_penalty = (1 - actual_l1_hit) * 100 + (1 - actual_l2_hit) * 300
            actual_time = base_time + cache_miss_penalty * 0.01
            
            results['l1_hit_rates'].append(actual_l1_hit)
            results['l2_hit_rates'].append(actual_l2_hit)
            results['execution_times'].append(actual_time)
            
            print(f"  {model}: L1命中{actual_l1_hit:.1%}, L2命中{actual_l2_hit:.1%}, 时间{actual_time:.1f}ms")
        
        total_time = max(results['execution_times'])
        print(f"  🕐 总执行时间: {total_time:.1f}ms (并发执行)")
        
        return results
    
    def calculate_random_cache_conflicts(self, setup):
        """
        计算随机分配导致的缓存冲突
        """
        # 模拟随机SM分配导致的缓存竞争
        total_cache_footprint = sum(setup['cache_footprints'])
        
        conflicts = {
            'l1_conflicts': [],
            'l2_conflicts': []
        }
        
        for i, footprint in enumerate(setup['cache_footprints']):
            # L1冲突：不同kernel随机分配到相同SM
            l1_conflict_prob = 0.6  # 60%概率发生L1冲突
            conflicts['l1_conflicts'].append(l1_conflict_prob)
            
            # L2冲突：总footprint超过L2容量
            l2_conflict_ratio = min(1.0, total_cache_footprint / 40)  # 40MB L2
            conflicts['l2_conflicts'].append(l2_conflict_ratio * 0.8)
            
        return conflicts
    
    def compare_results(self, random_results, controlled_results):
        """
        对比实验结果
        """
        print(f"\n📊 实验结果对比")
        print("=" * 30)
        
        random_total = max(random_results['execution_times'])
        controlled_total = max(controlled_results['execution_times'])
        speedup = random_total / controlled_total
        
        print(f"🎲 随机分配总时间: {random_total:.1f}ms")
        print(f"🎯 受控分配总时间: {controlled_total:.1f}ms")
        print(f"🚀 性能提升: {speedup:.1f}x")
        
        # Cache命中率对比
        print(f"\n💾 缓存性能对比:")
        print(f"  随机分配平均L1命中率: {sum(random_results['l1_hit_rates'])/3:.1%}")
        print(f"  受控分配平均L1命中率: {sum(controlled_results['l1_hit_rates'])/3:.1%}")
        print(f"  随机分配平均L2命中率: {sum(random_results['l2_hit_rates'])/3:.1%}")
        print(f"  受控分配平均L2命中率: {sum(controlled_results['l2_hit_rates'])/3:.1%}")
        
        # 结论
        print(f"\n💡 关键发现:")
        if speedup > 1.3:
            print(f"  ✅ 受控SM分配显著提升性能 (+{(speedup-1)*100:.0f}%)")
            print(f"  ✅ Cache局部性是性能关键因素")
            print(f"  ✅ 空分复用在受控分配下具备优势")
        else:
            print(f"  ⚠️ 受控分配提升有限 (+{(speedup-1)*100:.0f}%)")
            print(f"  ⚠️ 时分复用可能更加稳定")
